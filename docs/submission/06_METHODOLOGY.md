#### Law 1: Scope Check (ChromaDB Distance Scoring)

```python
def _check_scope(self, query: str) -> Dict[str, any]:
    results = self.vectorstore.similarity_search_with_score(
        query=query,
        k=3,
        filter={"course_id": self.course_id}
    )
    
    min_score = min(score for _, score in results)
    
    # Threshold empirically tuned:
    # - In-scope queries: 0.45-0.75
    # - Out-of-scope queries: 0.80+
    SCOPE_THRESHOLD = 0.80
    
    if min_score > SCOPE_THRESHOLD:
        return {"approved": False, "reason": "Topic not in COMP 237"}
    return {"approved": True}
```
